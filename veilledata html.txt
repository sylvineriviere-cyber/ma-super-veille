        // CALCULATEURS & CHARTS
        function updateTCO() {
            const el = document.getElementById('tcoTokens');
            if(!el) return;
            const v = parseInt(el.value);
            if(document.getElementById('tokenLabel')) document.getElementById('tokenLabel').innerText = `${v}M`;
            if(document.getElementById('costCloud')) document.getElementById('costCloud').innerText = (v * 45).toLocaleString() + ' €';
            if(document.getElementById('costSov')) document.getElementById('costSov').innerText = (v * 27).toLocaleString() + ' €';
        }
        if(document.getElementById('tcoTokens')) document.getElementById('tcoTokens').addEventListener('input', updateTCO);

        function updateGEO() {
            const el = document.getElementById('geoInput');
            if(!el) return;
            const v = parseInt(el.value);
            if(document.getElementById('geoValueDisplay')) document.getElementById('geoValueDisplay').innerText = v.toLocaleString();
            const l = Math.round(v * 0.25);
            const g = Math.round((v - l) * 0.15);
            if(document.getElementById('geoLoss')) document.getElementById('geoLoss').innerText = `-${l.toLocaleString()}`;
            if(document.getElementById('geoGain')) document.getElementById('geoGain').innerText = `+${g.toLocaleString()}`;
            if(document.getElementById('geoFinal')) document.getElementById('geoFinal').innerText = (v - l + g).toLocaleString();
        }
        if(document.getElementById('geoInput')) document.getElementById('geoInput').addEventListener('input', updateGEO);
